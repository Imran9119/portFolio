  ngOnInit(): void {
    this.showForm = false
    this.getAppList();
    this.state = history.state;
    this.SelectedTabIndex = this.state?.selectedTabIndex || 0;
    if (this.state && this.state.isEditMode === 'edit') {
      this.isEditMode = this.state.isCheckerRole ? false : true;
      this.availableName = true;
      this.availableName1 = true;
      this.availableDomain = true;
      this.isInCheckerQueue = this.state.isInCheckerQueue
      this.addUserList = this.state.addUserList || [];
      this.tempaddUserList = this.state.addUserList || [];
      this.groupEnableList = this.state.groupEnableList || [];
      this.tempgroupEnableList = this.state.groupEnableList || [];
      this.multipleCifList = this.state.multipleCifList || [];
      this.tempmultipleCifList = this.state.multipleCifList || [];
      this.rulesList = this.state.rulesList || [];
      this.temprulesList = this.state.rulesList || [];
      this.addGUserList = this.state.addGUserList || [];
      this.tempaddGUserList = this.state.addGUserList || [];

      const groupRoleMap = new Map<string, string>();

      (this.groupEnableList || []).forEach((g: any) => {
        if (g.groupName && g.groupRole) {
          groupRoleMap.set(g.groupName.trim().toLowerCase(), g.groupRole);
        }
      });

      this.addGUserList = (this.addGUserList || []).map((u: any) => {
        let roles: string[] = [];

        if (Array.isArray(u.groupName)) {
          roles = u.groupName
            .map((g: string) => groupRoleMap.get(g?.trim().toLowerCase()))
            .filter((r: any) => !!r); // remove null/undefined
        } else {
          const role = groupRoleMap.get(u.groupName?.trim().toLowerCase());
          roles = role ? [role] : [];
        }

        return {
          ...u,
          groupRole: roles
        };
      });

      this.addCustForm.patchValue({
        domainId: this.state.domainId,
        groupEnable: this.state.isGroupEnabled === 'Yes',
        multipleCif: this.state.isMultipleCif === 'Yes',
      })
      this.custType = this.state.custType || 'Corporate';
      if (this.custType === 'Retail') {
        this.addCustForm.get('users.groupName')?.disable();
        // this.addCustForm.get('users.groupUserRole')?.disable();
        this.addCustForm.get('users.fromLimit')?.disable();
        this.addCustForm.get('users.toLimit')?.disable();
        this.addCustForm.get('domainId')?.disable({ emitEvent: false, onlySelf: true });
        this.addCustForm.get('domainId')?.removeValidators([Validators.required]);
        this.addCustForm?.updateValueAndValidity()
      } else {
        this.addCustForm.get('domainId')?.disable({ emitEvent: false, onlySelf: true });
        // this.addCustForm.get('domainId')?.addValidators([Validators.required]);
        this.addCustForm.get('domainId')?.updateValueAndValidity();
        if (this.groupEnableList.length > 0) {
          this.addCustForm.get('rules.groupName')?.addValidators([Validators.required]);
          this.addCustForm.updateValueAndValidity();
        } else {
          this.addCustForm.get('rules.groupName')?.removeValidators([Validators.required]);
          this.addCustForm.updateValueAndValidity();
        }
      }
      this.domainId = this.state.domainId;
      this.cif = this.state.cifId;
      this.isCheckerRole = this.state.isCheckerRole;
      this.isDisableMode = false;
      this.isAddGroupButtonEnabled = this.state.isGroupEnabled !== 'Yes';
      this.previousData = {
        addCustForm: this.addCustForm.getRawValue(),
        multipleCifList: this.multipleCifList,
        groupEnableList: this.groupEnableList,
        addUserList: this.addUserList,
        addGUserList: this.addGUserList,
        rulesList: this.rulesList,
      }
      if (this.isCheckerRole) {
        if (this.multipleCifList.some((user: any) => user.isEditable === 1) || this.groupEnableList.some((user: any) => user.isEditable === 1)) {
          this.showCustomer = true
        } else {
          this.showCustomer = false
        }

        if (this.addUserList.some((user: any) => user.isEditable === 1) || this.addGUserList.some((user: any) => user.isEditable === 1)) {
          this.showUser = true
        } else {
          this.showUser = false
        }

        if (this.rulesList.some((user: any) => user.isEditable === 1)) {
          this.showRules = true
        } else {
          this.showRules = false
        }

      if (this.isInCheckerQueue) {
        this.notification.remove();

      }

      this.isEditMode = false
    }

    this.productType = this.state?.getMasterDataList?.productNames;
    this.productNames = Object.fromEntries(
      Object.entries(this.state?.getMasterDataList?.productNames ?? {}).map(
        ([key, value]) => [
          key,
          Array.isArray(value)
            ? value.map((item: any) => `${item.productDispName} - ${key}`)
            : []
        ]
      )

    if (this.isCheckerRole || this.isInCheckerQueue) {
      this.addCustForm.disable();
    } {
      this.onGroupEnableChange();
    }
    this.onValuesChanges();
    this.rulesValueChanges();
    this.onApproversCountChange()
    this._cdr.detectChanges();

      this.addCustForm.get('users.toLimit')?.clearValidators();
    } else {
      this.addCustForm.get('users.fromLimit')?.setValidators([Validators.required]);
      this.addCustForm.get('users.toLimit')?.setValidators([Validators.required]);
    }

  getAppList() {
    const data = {
      searchType: ''
    }
    this.service.getTwoFactorAuthDetails(data).subscribe((response: any) => {
      if (response.data.responseStatus.status === 'success') {
        this.isOverride = response.data.data.twoFactorAuthDetailsList[0].isOverride;
        if (this.isOverride === 0) {
          this.isTwoFactorEnabled = response.data.data.twoFactorAuthDetailsList[0].isTwoFactorEnabled;
          this.isEmail = response.data.data.twoFactorAuthDetailsList[0].isEmail;
          this.isSms = response.data.data.twoFactorAuthDetailsList[0].isSms;
          this.addCustForm.get('users.isTwoFactorEnabled')?.disable();
          this.addCustForm.get('users.isTwoFactorEnabled')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isTwoFactorEnabled === 1 ? true : false);
          this.addCustForm.get('users.isEmail')?.disable();
          this.addCustForm.get('users.isEmail')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isEmail === 1 ? true : false);
          this.addCustForm.get('users.isSms')?.disable();
          this.addCustForm.get('users.isSms')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isSms === 1 ? true : false);
          this.addCustForm.get('groupUser.isTwoFactorEnabled')?.disable();
          this.addCustForm.get('groupUser.isTwoFactorEnabled')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isTwoFactorEnabled === 1 ? true : false);
          this.addCustForm.get('groupUser.isEmail')?.disable();
          this.addCustForm.get('groupUser.isEmail')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isEmail === 1 ? true : false);
          this.addCustForm.get('groupUser.isSms')?.disable();
          this.addCustForm.get('groupUser.isSms')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isSms === 1 ? true : false);
        } else {
          this.addCustForm.get('users.isTwoFactorEnabled')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isTwoFactorEnabled === 1 ? true : false);
          this.addCustForm.get('users.   ')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isEmail === 1 ? true : false);
          this.addCustForm.get('users.isSms')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isSms === 1 ? true : false);
          this.addCustForm.get('groupUser.isTwoFactorEnabled')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isTwoFactorEnabled === 1 ? true : false);
          this.addCustForm.get('groupUser.isEmail')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isEmail === 1 ? true : false);
          this.addCustForm.get('groupUser.isSms')?.setValue(response.data.data.twoFactorAuthDetailsList[0].isSms === 1 ? true : false);
        }
      } else {
        this.notification.error('Error', response.responseStatus.message);
      }
    })
  }

  onCardClick(tabIndex: number): void {
    this.activeTabIndex = tabIndex;
  }

  editForm() {
    this.isDisableMode = false;
    this.addCustForm?.enable({ onlySelf: true, emitEvent: false });
    this.addCustForm?.get('domainId')?.disable({ onlySelf: true, emitEvent: false });
    this.addCustForm?.updateValueAndValidity({ onlySelf: true, emitEvent: false });
  }

  changeTab(e: NzTabChangeEvent[]) {
    this.showForm = false;
    this.isEditRuleEnable = false
    this.isEditGroupEnable = false;
    this.isEditAddUser = false;
    this.isEditGAddUser = false;
    if (this.activeTabIndex === 1 || this.activeTabIndex === 2) {
      const allowChange = this.multipleCifList.length > 0;
      if (!allowChange) {
        this.notification.remove()
        this.notification.warning('Alert', 'Please add CIF.')
        setTimeout(() => {
          this.activeTabIndex = 0;
        });
        this.SelectedTabIndex = 0;
      } else {
        this.SelectedTabIndex = this.activeTabIndex
      }
      this.rulesForm.reset();
      this.rulesForm.get('domainId')?.patchValue(this.addCustForm.get('domainId')?.value);
      this.rulesForm.updateValueAndValidity();
      this.checkerUserList = [];
      this.authorizerUserList = [];
      this.checkerGroupList = [];
      this.authorizerGroupList = [];
    } else {
      this.SelectedTabIndex = this.activeTabIndex
    }


    if (this.isEditMode || this.isCheckerRole) {
      this.isDisableMode = false;
      // this.addCustForm?.disable({ onlySelf: true, emitEvent: false });
      // this.addCustForm?.get('domainId')?.disable({ onlySelf: true, emitEvent: false });
      this.addCustForm?.updateValueAndValidity({ onlySelf: true, emitEvent: false });
    }
  }

  resetForm() {
    this.notification.remove();
    this.modal.closeAll();
    this.modal.warning({
      nzTitle: 'Confirm',
      nzContent: 'Changes will be lost. Please Confirm.',
      nzOkText: 'Yes',
      nzClosable: false,
      nzOkType: 'primary',
      nzOnOk: () => {
        switch (this.SelectedTabIndex) {
          case 0:
            this.multipleCifList = this.previousData.multipleCifList;
            this.groupEnableList = this.previousData.groupEnableList;
            this.deletedList.cifId = [];
            this.deletedList.groupName = [];

            break;

          case 1:
            this.addGUserList = this.previousData.addGUserList;
            this.addUserList = this.previousData.addUserList;
            this.deletedList.userId = []
            this.addCustForm.get('users')?.reset();
            this.addCustForm.get('groupUser')?.reset();
            break;

          case 2:
            this.rulesList = this.previousData.rulesList;
            this.deletedList.rules = []
            this.rulesForm.reset();
            break;
        }

        if (this.isEditMode && !this.isCheckerRole) {
          this.isDisableMode = false;
        }
        // this.addCustForm?.disable({ onlySelf: true, emitEvent: false });
        this.addCustForm?.updateValueAndValidity({ onlySelf: true, emitEvent: false });
        this.showForm = false;
        this.isEditAddUser = false;
        this.isEditGAddUser = false;
        this.isEditRuleEnable = false;
        this.isEditGroupEnable = false;
        this._cdr.detectChanges();
      },
      nzCancelText: 'No',
    });
  }

  onValuesChanges() {
    this.addCustForm.get('groupEnable')?.valueChanges.subscribe((res: boolean) => {
      if (this.groupEnableList && this.groupEnableList.length > 0) {
        this.addCustForm.get('groupEnable')?.patchValue(true, { emitEvent: false, onlySelf: true });
        this.notification.remove();
        this.modal.closeAll();

          nzTitle: 'Confirm',
          nzContent: 'Changes will be lost. Please Confirm.',
          nzOkText: 'Yes',
          nzClosable: false,
          nzOkType: 'primary',

            this.groupEnableList = [];

            this.addCustForm.get('users')?.patchValue({
              groupName: '',

                this.addUserList = [];
                this.groupEnableList = [];
                this.rulesList = [];

              nzOnCancel: () => {
                this.addCustForm.get('multipleCif')?.patchValue(false, { emitEvent: false, onlySelf: true });

            this.addCustForm.get('domainId')?.disable({ emitEvent: false, onlySelf: true });

  onFromLimitBlur(groupdetails: any): void {
    const fromCtrl = this.addCustForm.get('rules.fromLimit');
    const toCtrl = this.addCustForm.get('rules.toLimit');

    if (!fromCtrl?.value) {
      return;
    }

    const fromLimit = parseFloat(fromCtrl.value);
    const toLimit = parseFloat(toCtrl?.value);
    const groupFrom = parseFloat(groupdetails?.fromLimit);
    const groupTo = parseFloat(groupdetails?.toLimit);

    if (groupdetails && fromLimit < groupFrom) {
      fromCtrl.patchValue(0, { emitEvent: false, onlySelf: true });
      fromCtrl.setErrors({ min: true });
      this.notification.remove();
      this.notification.warning('Alert', 'From Limit cannot be less than group From Limit.');
      return;
    }

    if (groupdetails && fromLimit > groupTo) {
      fromCtrl.patchValue(0, { emitEvent: false, onlySelf: true });
      fromCtrl.setErrors({ min: true });
      this.notification.remove();
      this.notification.warning('Alert', 'From Limit cannot be greater than group To Limit.');
      return;
    }

    if (toCtrl?.value && fromLimit > toLimit) {
      fromCtrl.patchValue(0, { emitEvent: false, onlySelf: true });
      fromCtrl.setErrors({ max: true });
      this.notification.remove();
      this.notification.warning('Alert', 'From Limit cannot be greater than To Limit.');
      return;
    }

    fromCtrl.setErrors(null);
  }

  onToLimitBlur(groupdetails: any): void {
    const toCtrl = this.addCustForm.get('rules.toLimit');
    const fromCtrl = this.addCustForm.get('rules.fromLimit');

    if (!toCtrl?.value) {
      return;
    }

    const toLimit = parseFloat(toCtrl.value);
    const fromLimit = parseFloat(fromCtrl?.value);
    const groupTo = parseFloat(groupdetails?.toLimit);
    const groupFrom = parseFloat(groupdetails?.fromLimit);

    if (groupdetails && toLimit > groupTo) {
      toCtrl.patchValue(0, { emitEvent: false, onlySelf: true });
      toCtrl.setErrors({ max: true });
      this.notification.remove();
      this.notification.warning('Alert', 'To Limit cannot be greater than group To Limit.');
      return;
    }

    if (groupdetails && fromLimit < groupFrom) {
      toCtrl?.patchValue(0, { emitEvent: false, onlySelf: true });
      toCtrl?.setErrors({ min: true });
      this.notification.remove();
      this.notification.warning('Alert', 'To Limit cannot be less than group From Limit.');
      return;
    }

    if (fromCtrl?.value && toLimit < fromLimit) {
      toCtrl.patchValue(0, { emitEvent: false, onlySelf: true });
      toCtrl.setErrors({ min: true });
      this.notification.remove();
      this.notification.warning('Alert', 'To Limit cannot be less than From Limit.');
      return;
    }

    if (fromCtrl?.value && toLimit === fromLimit) {
      toCtrl.patchValue(0, { emitEvent: false, onlySelf: true });
      toCtrl.setErrors({ min: true });
      this.notification.remove();
      this.notification.warning('Alert', 'To Limit cannot be same as From Limit.');
      return;
    }

    toCtrl.setErrors(null);
  }

  searchCifData(cifId: string, actionType: string) {
    if (this.addCustForm.get('cifs')?.invalid && !this.isEditMultipleCif) {
      this.addCustForm.get('cifs.cifId')?.markAsDirty();
      this.addCustForm.get('cifs.cifId')?.updateValueAndValidity();
      this.notification.remove();
      this.notification.warning('Alert', 'Please Enter Cif.');
      return;
    }
    let isDuplicate = false;
    this.multipleCifList.forEach((res: any) => {
      if (res.cifId == this.addCustForm.get('cifs.cifId')?.value) {
        isDuplicate = true
      }
    })
    if (isDuplicate) {
      this.notification.remove();
      this.notification.warning('Alert', 'Duplicate Cif is not allowed to be added.');
      return;
    } else if (!this.addCustForm.get('multipleCif')?.value && this.multipleCifList.length == 1 && !this.isEditMultipleCif) {
      this.notification.remove();
      this.notification.warning('Alert', 'Please Enable multiple Cif.');
      return;
    }
    this.isSearched = true;
    let listOfUserRoleLocation: UserRoleLocation[] = []
    this.store.select(selectUserRoleLocation).subscribe((res: UserRoleLocation[]) => {
      if (res) {
        listOfUserRoleLocation = res;
      }
    })
    const payload: SearchCifData = {
      cifId: cifId,
      actionType: actionType
    };
    if (actionType === 'add') {
      payload.listOfUserRoleLocation = listOfUserRoleLocation
    }

    this.service.getAccDtls(payload).subscribe((res: any) => {
      if (res.data.responseStatus.status === 'success') {
        const custInfoObj = res.data.data.custInfoObj;
        this.customerId = res.data.data.cif;
        this.custType = res.data.data.custType;
        if (this.custType === 'Retail') {
          this.addCustForm.get('users.groupName')?.disable();
          // this.addCustForm.get('users.groupUserRole')?.disable();
          this.addCustForm.get('users.fromLimit')?.disable();
          this.addCustForm.get('users.toLimit')?.disable();
          this.addCustForm.get('domainId')?.disable({ emitEvent: false, onlySelf: true });
          this.addCustForm.get('domainId')?.removeValidators([Validators.required]);
          if (this.addCustForm.get('multipleCif')?.value) {
            this.isCifSearchSucess = false;
            this.notification.remove();
            this.notification.warning('Alert', 'Multiple CIFs with Retail Customer Type cannot be added.');
          } else {
            this.groupEnableList = [];
            this.rulesList = [];
            this.isCifSearchSucess = true;
          }
        } else {
          this.isCifSearchSucess = true;
          this.addCustForm.get('users.groupName')?.enable();
          this.addCustForm.get('users.groupUserRole')?.enable();
          this.addCustForm.get('users.fromLimit')?.enable();
          this.addCustForm.get('users.toLimit')?.enable();
          if (!this.isEditMode) {
            this.addCustForm.get('domainId')?.enable({ emitEvent: false, onlySelf: true });
            this.addCustForm.get('domainId')?.addValidators([Validators.required]);
          }
        }

        if (this.isCifSearchSucess) {
          this.addCustForm.get('cifs')?.patchValue({
            custName: custInfoObj?.customerName?.toUpperCase(),
            corporateId: custInfoObj?.corpId,
            ieCode: custInfoObj?.ieCode,
          });
          this.accDtlsModal(res?.data?.data);
        }
        this.addCustForm?.updateValueAndValidity();
      } else {
        this.isCifSearchSucess = false;
        this.notification.remove();
        this.notification.warning('Alert', res.data.responseStatus.message);
      }
    });
  }

  addGroupEnable() {
    const groupEnable = this.addCustForm.get('groups') as FormGroup;

    if (groupEnable?.valid) {
      if (parseFloat(groupEnable?.get('fromLimit')?.value) > parseFloat(groupEnable?.get('toLimit')?.value)) {
        this.notification.remove();
        this.notification.warning('Alert', 'To Limit cannot be less than From Limit.');
        return;
      }

      if (parseFloat(groupEnable?.get('fromLimit')?.value) === parseFloat(groupEnable?.get('toLimit')?.value)) {
        this.notification.remove();
        this.notification.warning('Alert', 'To Limit cannot be same as From Limit.');
        return;
      }

      const cifPayload = {
        groupName: groupEnable.get('groupName')?.value,
        groupRole: groupEnable.get('groupRole')?.value,
        cifId: groupEnable.get('cifId')?.value,
        groupDescription: groupEnable.get('groupDescription')?.value,
        fromLimit: groupEnable.get('fromLimit')?.value,
        toLimit: groupEnable.get('toLimit')?.value,
        selectedAccounts: groupEnable.get('selectedAccounts')?.value
      };
      if (this.isEditGroupEnable && this.editedGroupEnabledIndex !== null) {
        const updatedList = [...this.groupEnableList];
        updatedList[this.editedGroupEnabledIndex] = { ...cifPayload, isEnabled: this.groupEnableList[this.editedGroupEnabledIndex].isEnabled };
        this.groupEnableList = updatedList;
      } else {
        const exists = this.groupEnableList?.some(r => r.groupName === cifPayload.groupName);
        if (exists) {
          this.notification.remove();
          this.notification.warning('Alert', 'Duplicate records are not allowed to be added.');
          return;
        }
        this.groupEnableList = [...this.groupEnableList, { ...cifPayload, isEnabled: true }];
      }
      this.resetGroupEnableForm();
      this.addCustForm.get('groups.groupName')?.enable();
      this.addCustForm.get('rules.groupName')?.addValidators([Validators.required]);
      this.addCustForm.updateValueAndValidity();
    } else {
      Object.values(groupEnable.controls).forEach(control => {
        if (control.invalid) {
          control.markAsDirty();
          control.updateValueAndValidity({ onlySelf: true });
        }
      });
      this.notification.remove();
      this.notification.warning('Alert', 'Please fill all mandatory data.');
    }
  }

  addUser() {

    if (this.custType === 'Retail') {
      this.addCustForm.get('users.groupUserRole')?.disable({ emitEvent: false, onlySelf: true });
      this.addCustForm.get('users')?.updateValueAndValidity()
    }
    const addUserControl = this.addCustForm.get('users') as FormGroup;
    const usersFormValue = this.addCustForm.get('users')?.getRawValue();
    if (this.availableName) {
      if (addUserControl?.valid) {
        if (parseFloat(usersFormValue.fromLimit) > parseFloat(usersFormValue.toLimit)) {
          this.notification.remove();
          this.notification.warning('Alert', 'To Limit cannot be less than From Limit.');
          return;
        }
        if (parseFloat(usersFormValue.fromLimit) === parseFloat(usersFormValue.toLimit)) {
          this.notification.remove();
          this.notification.warning('Alert', 'To Limit cannot be same as From Limit.');
          return;
        }
        const subProductAccess: any[] = usersFormValue.subProductAccess || [];
        const cleanedSubProductAccess = subProductAccess.map(item => item.split(' - ')[0].trim());
        const allProducts: any[] = Object.entries(this.productType).flatMap(([category, products]: [string, any]) =>
          products.map((product: any) => ({ ...product, category }))
        );
        const matchedProductTypes: string[] = allProducts
          .filter(product => cleanedSubProductAccess.includes(product.productDispName))
          .map(product => `${product.productType} - ${product.category}`);
        const addUserPayload = {
          userId: usersFormValue.userId,
          userFName: usersFormValue.userFName,
          userLName: usersFormValue.userLName,
          mobileNo: usersFormValue.mobileNo,
          emailId: usersFormValue.emailId,
          cifId: usersFormValue.cifId,
          accountNo: usersFormValue.accountNo,
          // groupName: usersFormValue.groupName,
          fromLimit: usersFormValue.fromLimit,
          toLimit: usersFormValue.toLimit,
          groupUserRole: usersFormValue.groupUserRole,
          productAccess: usersFormValue.productAccess,
          subProductAccess: usersFormValue.subProductAccess,
          subProductTypeAccess: matchedProductTypes,
          isTwoFactorEnabled: usersFormValue.isTwoFactorEnabled === true && this.isOverride === 0 && this.isTwoFactorEnabled === 2 ? 2 : usersFormValue.isTwoFactorEnabled === true && this.isOverride === 0 && this.isTwoFactorEnabled === 1 ? 1 : usersFormValue.isTwoFactorEnabled === true ? 1 : 0,
          isEmail: usersFormValue.isEmail === true && this.isOverride === 0 && this.isEmail === 2 ? 2 : usersFormValue.isEmail === true && this.isOverride === 0 && this.isEmail === 1 ? 1 : usersFormValue.isEmail === true ? 1 : 0,
          isSms: usersFormValue.isSms === true && this.isOverride === 0 && this.isSms === 2 ? 2 : usersFormValue.isSms === true && this.isOverride === 0 && this.isSms === 1 ? 1 : usersFormValue.isSms === true ? 1 : 0,
        };
        if (usersFormValue.isTwoFactorEnabled === true && usersFormValue.isEmail === false && usersFormValue.isSms === false) {
          this.notification.info('Information', 'Two Factor Authentication is enabled but both Email and SMS options are disabled. Please enable at least one option for Two Factor Authentication to work.');

          const currentData = this.addCustForm.get('users')?.getRawValue();

            if (JSON.stringify(currentData) === JSON.stringify(this.originalEditUserData)) {
              this.notification.remove();
              this.notification.info('Information', 'No changes done.');
              return;
            }
          const updatedList = [...this.addUserList];
          updatedList[this.editedAddUserIndex] = { ...addUserPayload, isEnabled: this.addUserList[this.editedAddUserIndex].isEnabled };
          this.addUserList = updatedList;
        } else {
          const errors: string[] = [];

          if (this.addUserList?.some(r => r.userId === addUserPayload.userId)) {
            errors.push('User ID already exists.');
          }

          if (this.addUserList?.some(r => r.emailId === addUserPayload.emailId)) {
            errors.push('Email ID already exists.');
          }

            errors.push('Mobile number already exists.');

          isTwoFactorEnabled: this.isTwoFactorEnabled === '1' || this.isTwoFactorEnabled === 1 ? true : false,
          isEmail: this.isEmail === '1' || this.isEmail === 1 ? true : false,
          isSms: this.isSms === '1' || this.isSms === 1 ? true : false,
        });
      } else {
        Object.values(addUserControl.controls).forEach(control => {
          if (control.invalid) {
            control.markAsDirty();
            control.updateValueAndValidity({ onlySelf: true });
          }
        });
        this.onGroupEnableChange();
        this.notification.remove();
        this.notification.warning('Alert', 'Please fill all mandatory data.');
      }
    } else {
      this.notification.remove();
      this.notification.error('Error', 'User Id details already present');

    this.notification.remove();
    const rulesGroup = this.addCustForm.get('rules') as FormGroup;

    // if(this.addCustForm.get('rules')?.invalid) {

    if (rulesGroup?.valid) {
      let authMatrix = '';
      if (parseInt(rulesFormValue.checkerCount) > 0) {
        authMatrix += rulesFormValue.checkerCount + 'C';
      }
      if (parseInt(rulesFormValue.authorizerCount) > 0) {
        if (parseInt(rulesFormValue.checkerCount) > 0) {
          authMatrix += '+' + rulesFormValue.authorizerCount + 'A';
        } else {
          authMatrix += rulesFormValue.authorizerCount + 'A';
        }
      }
      // const groupRoles = this.groupEnableList.length > 0 ? this.groupEnableList.find(group => group.groupName === rulesFormValue.groupName)?.groupRole
      //   : [
      //     ...new Set(

      //   if (authMatrix === '') {
      //     authMatrix = '1M'
      //   } else {
      //     authMatrix = '1M+' + authMatrix;
      //   }
      // }

      const groupRoles = this.groupEnableList.length > 0
        ? [
          ...new Set(
            this.groupEnableList
              .filter(group => rulesFormValue.groupName?.includes(group.groupName))
              .flatMap(group => group.groupRole || [])
          )
        ]
        : [
          ...new Set(
            this.addUserList.flatMap(item => item.groupUserRole ? item.groupUserRole : [])
          )
        ];

      if (groupRoles?.includes('MAKER')) {
        if (authMatrix === '') {
          authMatrix = '1M';
        } else {
          authMatrix = '1M+' + authMatrix;
        }
      }

      if (authMatrix === '') {
        rulesGroup?.get('checkerCount')?.setErrors({ required: true });
        rulesGroup?.get('authorizerCount')?.setErrors({ required: true });
        this.notification.remove();
        this.notification.warning('Alert', 'Auth Matrix cannot be empty. Please specify Checker Count and/or Authorizer Count.');
        return;
      }
      const rulePayload: Rules = {
        groupName: rulesFormValue.groupName,
        authMatrix: authMatrix,
        // fromLimit: rulesFormValue.fromLimit,
        // toLimit: rulesFormValue.toLimit,

      }

      if (validationMessage) {
        this.notification.remove();

        this.markRulesInvalidFieldsDirty();
        return;
      }

      const exists = this.rulesList?.some((r: any) =>
        r.groupName === rulesFormValue.groupName &&
        r.authMatrix === rulesFormValue.authMatrix
        // r.fromLimit === rulesFormValue.fromLimit &&
        // r.toLimit === rulesFormValue.toLimit &&
      );
      // if (this.addCustForm.get('groupEnable')?.value && this.groupEnableList.length > 0) {

      //     user.groupName?.some((group: any) =>

      //   ).length;
      //     if ((groupUsersCount < (parseInt(rulesFormValue.checkerCount) + parseInt(rulesFormValue.authorizerCount)))) {

      //     this.notification.warning('Alert', 'Rule cannot be added as the number of users in the group is less than the total number of checkers and authorizers specified.');
      //     return;
      //   }
      // }

      if (this.addCustForm.get('groupEnable')?.value && this.groupEnableList.length > 0) {

        const selectedGroups = new Set(rulesFormValue.groupName || []);

        const groupUsersCount = this.addGUserList.filter((user: any) => {
          const userGroups = Array.isArray(user.groupName)
            ? user.groupName
            : [user.groupName];

          return user.groupName.some((g: string) => selectedGroups.has(g));
        }).length;

        const checkerCount = parseInt(rulesFormValue.checkerCount || '0');
        const authorizerCount = parseInt(rulesFormValue.authorizerCount || '0');
        const requiredUsers = checkerCount + authorizerCount;

        if (groupUsersCount < requiredUsers) {
          this.notification.remove();
          this.notification.warning(
            'Alert',
            'Rule cannot be added as the number of users in the group is less than the total number of checkers and authorizers specified.'
          );
          return;
        }
      }

      // this.onToLimitBlur(this.groupdetails);
      // this.onFromLimitBlur(this.groupdetails);
      if (this.isEditRuleEnable && this.editedRuleIndex !== null) {

        updatedList[this.editedRuleIndex] = { ...rulePayload, isEnabled: this.rulesList[this.editedRuleIndex].isEnabled };

        this.resetRulesForm();

          return;
        } else {
          this.rulesList = [...this.rulesList, { ...rulePayload, isEnabled: true }];
          this.isruleTab = false;
          this.resetRulesForm();
        }
      }
    } else {
      this.notification.remove();
      this.notification.warning('Alert', 'Please fill all mandatory data.');
      this.markRulesInvalidFieldsDirty();
    }

  getKey(item: any, type: 'GROUP' | 'CIF' | 'RULE'): string {
    switch (type) {
      case 'GROUP':
        return item.groupName?.trim().toLowerCase();

      case 'CIF':
        return typeof item === 'object'
          ? item.cifId?.toString()
          : item?.toString();

      case 'RULE':
        return item.ruleName || JSON.stringify(item);

      default:
        return JSON.stringify(item);
    }
  }
  addActionTypeToRest(
    updatedList: any[],
    tempList: any[],
    type: 'GROUP' | 'CIF' | 'RULE'
  ): any[] {

    if (!updatedList?.length) return [];

    const initialMap = new Map<string, any>();

    (tempList || []).forEach((item: any) => {
      const key = this.getKey(item, type);
      initialMap.set(key, item);
    });

    return updatedList.map((item: any) => {
      const key = this.getKey(item, type);

      return {
        ...item,
        actionType: initialMap.has(key) ? 'Update' : 'Add'
      };
    });
  }

  getUpdatedValues(previousData: any, currentData: any) {
    return this.getChangedValues(previousData, currentData);
  }

  getChangedValues(oldObj: any, newObj: any): any {

    // same reference / same primitive
    if (oldObj === newObj) return {};

    // ignore undefined
    if (newObj === undefined) return {};

    // ================= ARRAY HANDLING =================
    if (Array.isArray(newObj)) {
      const oldArray = Array.isArray(oldObj) ? oldObj : [];

      // empty arrays
      if (oldArray.length === 0 && newObj.length === 0) {
        return {};
      }

      // check if object array
      const firstItem = newObj[0];

      if (
        firstItem &&
        typeof firstItem === 'object' &&
        !Array.isArray(firstItem)
      ) {

        const keyField = this.getArrayKey(firstItem);

        // compare using unique key
        if (keyField) {
          const oldMap = new Map<string, any>();

          oldArray.forEach((item: any) => {
            oldMap.set(String(item[keyField]), item);
          });

          const changedItems: any[] = [];

          for (const item of newObj) {
            const key = String(item[keyField]);
            const oldItem = oldMap.get(key);

            const diff = this.getChangedValues(oldItem, item);

            const isChanged =
              (typeof diff === 'object' &&
                diff !== null &&
                Object.keys(diff).length > 0) ||
              typeof diff !== 'object';

            if (isChanged) {
              changedItems.push(item);
            }
          }

          return changedItems.length ? changedItems : {};
        }
      }

      // primitive arrays compare
      if (JSON.stringify(oldArray) !== JSON.stringify(newObj)) {
        return newObj;
      }

      return {};
    }

    // ================= OBJECT HANDLING =================
    if (typeof newObj === 'object' && newObj !== null) {

      const changes: any = {};

      const ignoredKeys = [
        'subProductTypeAccess'
      ];

      const keys = new Set([
        ...Object.keys(oldObj || {}),
        ...Object.keys(newObj || {})
      ]);

      keys.forEach((key: string) => {

        if (ignoredKeys.includes(key)) return;

        const oldValue = oldObj ? oldObj[key] : undefined;
        const newValue = newObj ? newObj[key] : undefined;

        if (newValue === undefined) return;

        const diff = this.getChangedValues(oldValue, newValue);

        const isChanged =
          (typeof diff === 'object' &&
            diff !== null &&
            Object.keys(diff).length > 0) ||
          typeof diff !== 'object';

        if (isChanged) {
          changes[key] = diff;
        }
      });

      return Object.keys(changes).length ? changes : {};
    }

    // ================= PRIMITIVE =================
    return oldObj !== newObj ? newObj : {};
  }

  maskMobileNumber(mobile: string): string {
    if (!mobile || mobile.length < 4) {
      return mobile;
    }

    const firstTwo = mobile.substring(0, 2);
    const lastTwo = mobile.substring(mobile.length - 2);

    return firstTwo + '******' + lastTwo;
  }

  maskEmail(email: string): string {
    if (!email) return '';

    const [name, domain] = email.split('@');
    if (!name || !domain) return email;

    const visiblePart1 = name.charAt(0);
    const visiblePart2 = name.charAt(name.length - 1);
    const maskedId = '******';

    const [domainName, extension] = domain.split('.');
    const firstLetter = domainName.charAt(0);
    const lastLetter = domainName.charAt(domainName.length - 1);
    const maskedDomain = `${firstLetter}******${lastLetter}`;

    return `${visiblePart1}${maskedId}${visiblePart2}@${maskedDomain}.${extension}`;
  }

    if (
      (this.filtercheckerUsers() === 0 && this.checkerUserList.length === 0) ||

  updateUsersAndGroups(
    type: 'DOMAIN' | 'CIF' | 'ACCOUNT',
    value?: any[]
  ) {
    this.checkerGroupList = [];
    this.authorizerGroupList = [];
    this.checkerUserList = [];
    this.authorizerUserList = [];

    //  GROUP FILTERING
    if (this.groupEnableList?.length > 0) {
      this.groupEnableList.forEach((group: any) => {
        let match = false;

        if (type === 'DOMAIN') {
          match = true;
        } else if (type === 'CIF') {
          match = group.cifId?.some((id: any) => value?.includes(id));
        } else if (type === 'ACCOUNT') {
          match = group.selectedAccounts?.some((id: any) => value?.includes(id));
        }

        if (match) {

          //  check if this group actually has users mapped
          const hasUsersInGroup = this.addGUserList?.some((u: any) => {
            const groupNames = Array.isArray(u.groupName)
              ? u.groupName
              : [u.groupName];

            return groupNames.some(
              (g: string) =>
                g?.trim().toLowerCase() === group.groupName?.trim().toLowerCase()
            );
          });

          if (!hasUsersInGroup) return; //  skip unused groups

          if (group.groupRole?.includes('CHECKER') &&
            !this.checkerGroupList.includes(group.groupName)) {
            this.checkerGroupList.push(group.groupName);
          }

          if (group.groupRole?.includes('AUTHORIZER') &&
            !this.authorizerGroupList.includes(group.groupName)) {
            this.authorizerGroupList.push(group.groupName);
          }
        }
      });
    }

    // ✅ USER FILTERING
    if (this.addUserList?.length > 0) {
      const valueSet = new Set(value || []);

      this.addUserList.forEach((user: any) => {
        let match = false;

        if (type === 'DOMAIN') {
          match = true;
        } else if (type === 'CIF') {
          const userCifList = Array.isArray(user.cifId)
            ? user.cifId
            : typeof user.cifId === 'string'
              ? user.cifId.split(',').map((id: string) => id.trim())
              : [];

          match = userCifList.some((id: any) => valueSet.has(id));
        } else if (type === 'ACCOUNT') {
          match = user.accountNo?.some((id: any) => valueSet.has(id));
        }

        if (!match) return;

        if (user.groupUserRole?.includes('CHECKER')) {
          this.checkerUserList.push(user.userId);
        }

        if (user.groupUserRole?.includes('AUTHORIZER')) {
          this.authorizerUserList.push(user.userId);
        }
      });
    }
  }

  filterAuthorizerUsers(): number {
    if (!this.authorizerGroupList?.length) return 0;

    const groupSet = new Set(
      this.authorizerGroupList.map(g => g.trim().toLowerCase())
    );

    const userSet = new Set<string>();

    (this.addGUserList || []).forEach((user: any) => {
      const hasMatch = this.normalizeToArray(user.groupName).some((g: string) =>
        groupSet.has(g?.trim().toLowerCase())
      );

      if (hasMatch) {
        userSet.add(user.userId);
      }
    });

    return userSet.size;
  }

  submitRule() {
    const hasMaker =
      this.addUserList?.some((u: any) =>
        u.groupUserRole?.includes('MAKER')
      ) ||
      this.addGUserList?.some((g: any) =>
        g.groupRole === 'MAKER'
      );

    if (!hasMaker && !this.isCheckerRole) {
      this.notification.remove();
      this.notification.warning('Alert', 'At least 1 Maker is required');
      return;
    }
    const hasChecker = (this.checkerUserList.length > 0 || this.checkerGroupList.length > 0) && this.rulesForm.get('checkerCount')?.value > 0;


    if (!hasChecker && !this.isCheckerRole) {
      this.notification.remove();
      this.notification.warning('Alert', 'At least 1 Checker is required');
      return;
    }

    if (!this.validateUserCombination()) {
      return;
    }

    if (this.rulesForm.get('setRuleOn')?.value === 'CIF') {
      this.rulesForm.get('cifId')?.addValidators([Validators.required]);
      this.rulesForm.get('cifId')?.updateValueAndValidity();
    } else {  
      this.rulesForm.get('cifId')?.clearValidators();
      this.rulesForm.get('cifId')?.updateValueAndValidity();
    }
    if (this.rulesForm.get('setRuleOn')?.value === 'DOMAIN') {
       this.rulesForm.get('domainId')?.addValidators([Validators.required]);
      this.rulesForm.get('domainId')?.updateValueAndValidity();
    } else {
      this.rulesForm.get('domainId')?.clearValidators();
      this.rulesForm.get('domainId')?.updateValueAndValidity();
    }
    if (this.rulesForm.get('setRuleOn')?.value === 'ACCOUNT') {
      this.rulesForm.get('accounts')?.addValidators([Validators.required]);
      this.rulesForm.get('accounts')?.updateValueAndValidity();
    } else {
      this.rulesForm.get('accounts')?.clearValidators();
      this.rulesForm.get('accounts')?.updateValueAndValidity();
    }

    if (this.rulesForm.valid) {
      const rulesFormValue = this.rulesForm?.getRawValue();

      let authMatrix = '1M';
      if (parseInt(rulesFormValue.checkerCount) > 0) {
        authMatrix += '+' + rulesFormValue.checkerCount + 'C';
      }
      if (parseInt(rulesFormValue.authorizerCount) > 0) {
        if (parseInt(rulesFormValue.checkerCount) > 0) {
          authMatrix += '+' + rulesFormValue.authorizerCount + 'A';
        } else {
          authMatrix += rulesFormValue.authorizerCount + 'A';
        }
      }
      const rulePayload = {
        ...rulesFormValue,
        authMatrix: authMatrix
      }

      const errorArr: string[] = []
      const exist = this.rulesList?.filter((r: any) =>
        r.setRuleOn === rulesFormValue.setRuleOn &&
       ( r.cifId === rulesFormValue.cifId ||
        r.accounts === rulesFormValue.accounts ||
        r.domainId === rulesFormValue.domainId )
      )

      if (exist.length > 0) {
        errorArr.push(`Rule authorization is already added for this ${rulesFormValue.setRuleOn}.`)
      }
      else {
        if (this.rulesList?.some((r: any) =>
          r.ruleName === rulesFormValue.ruleName)) {
          errorArr.push('Rule Name already added.')
        }
        // const exists = this.rulesList?.filter((r: any) =>
        //   r.authMatrix === authMatrix &&
        //   r.setRuleOn === rulesFormValue.setRuleOn &&
        //   r.cifId === rulesFormValue.cifId &&
        //   r.accounts === rulesFormValue.accounts &&
        //   r.domainId === rulesFormValue.domainId
        // )

        // if (exists.length > 0 && exist.length === 0){
        //   errorArr.push(`Same rule is added to rule : ${exists[0].ruleName}.`)
        // }
      }

      if (this.isEditRuleEnable && this.editedRuleIndex !== null) {
        const currentData = this.rulesForm?.getRawValue();
        if (JSON.stringify(currentData) === JSON.stringify(this.originalEditRuleData)) {
          this.notification.remove();
          this.notification.info('Information', 'No changes done.');
          return;
        }
        const updatedList = [...this.rulesList];
        updatedList[this.editedRuleIndex] = { ...rulePayload, isEnabled: this.rulesList[this.editedRuleIndex].isEnabled };
        this.rulesList = updatedList;
        this.resetRules();
        this.isruleTab = false;
      } else {
        if (errorArr.length) {
          const message = errorArr
            .map((msg, index) => `${index + 1}. ${msg}`)
            .join('<br/>');
          this.notification.remove();
          this.notification.warning('Alert', message, {
            nzDuration: 3000,
            nzPauseOnHover: true
          });
          return;
        } else {
          this.rulesList = [...this.rulesList, { ...rulePayload, isEnabled: true }];
          this.resetRules();
          this.isruleTab = false;
        }
      }
    } else {
      Object.values(this.rulesForm.controls).forEach(control => {
        if (control.invalid) {
          control.markAsDirty();
          control.updateValueAndValidity({ onlySelf: true });
        }
      });
      this.notification.remove();
      this.notification.warning('Alert', 'Please fill all mandatory data.');
    }
    this._cdr.detectChanges();
  }

      const checkerMatch = authMatrix.match(/(\d+)\s*C/);
      const authorizerMatch = authMatrix.match(/(\d+)\s*A/);

      checkerCount = checkerMatch ? parseInt(checkerMatch[1], 10) : 0;
      authorizerCount = authorizerMatch ? parseInt(authorizerMatch[1], 10) : 0;
    }

    this.rulesForm?.patchValue({
      ruleName: data.ruleName,
      authorizerCount: authorizerCount,
      checkerCount: checkerCount,
      accounts: data.accounts,
      cifId: data.cifId,
      domainId: data.domainId,
      isEnabled: data.isEnabled,
      setRuleOn: data.setRuleOn,
      authorizers: data.authorizers,
      checkers: data.checkers
    }, { emitEvent: false });
    this.originalEditRuleData = this.rulesForm?.getRawValue();
    switch (data.setRuleOn) {
      case 'DOMAIN':
        this.updateUsersAndGroups('DOMAIN');
        break;

      case 'CIF':
        this.updateUsersAndGroups('CIF', data.cifId);
        break;

      case 'ACCOUNT':
        this.updateUsersAndGroups('ACCOUNT', data.accounts);
        break;
    }


    // rebuild form arrays
    this.setCheckers(checkerCount);
    this.setAuthorizers(authorizerCount);
    this.validateUserCombination();
    this.checkers.clear();
    this.authorizers.clear();

    // ✅ Patch Checkers
    (data.checkers || []).forEach((chk: any, index: number) => {
      this.checkers.push(
        new FormGroup({
          checker: new FormControl(index + 1),
          groupList: new FormControl(chk.groupList || []),
          userList: new FormControl(chk.userList || [])
        })
      );
    });

    // ✅ Patch Authorizers
    (data.authorizers || []).forEach((auth: any, index: number) => {
      this.authorizers.push(
        new FormGroup({
          authorizer: new FormControl(index + 1),
          groupList: new FormControl(auth.groupList || []),
          userList: new FormControl(auth.userList || [])
        })
      );
    });
    this.rulesForm.get('ruleName')?.disable();
    this.originalEditRuleData = this.rulesForm?.getRawValue();

  }

  validateUserCombination(): boolean {
    const checker = this.rulesForm.get('checkerCount')?.value || 0;
    const authorizer = this.rulesForm.get('authorizerCount')?.value || 0;

    //  UNIQUE USERS 
    const uniqueUsers = new Set<string>();

    console.log(this.addUserList);
    console.log(this.addGUserList);

    const makerUsers = [
      // Individual users
      ...(this.addUserList || [])
        .filter((u: any) => u.groupUserRole?.includes('MAKER'))
        .map((u: any) => ({
          userId: u.userId,
          roles: u.groupUserRole
        })),

      // Group users
      ...(this.addGUserList || [])
        .filter((g: any) => g.groupRole?.includes('MAKER'))
        .map((g: any) => ({
          userId: g.userId,
          roles: g.groupRole // normalize
        }))
    ];
    const singleMakerWithExtraRole =
      makerUsers.length === 1 &&
      makerUsers[0].roles.length > 1;

    const invalidUserId = singleMakerWithExtraRole
      ? makerUsers[0].userId
      : null;


    // Individual users
    if (this.hasOnlyMaker()) {
      (this.checkerUserList || []).forEach((u: any) => uniqueUsers.add(u));
      (this.authorizerUserList || []).forEach((u: any) => uniqueUsers.add(u));
    } else {
      (this.checkerUserList || []).forEach((id: any) => {
        if (id !== invalidUserId) {
          uniqueUsers.add(id);
        }
      });

      (this.authorizerUserList || []).forEach((id: any) => {
        if (id !== invalidUserId) {
          uniqueUsers.add(id);
        }
      });
    }






    // Group users
    const checkerGroupUsers = this.getCheckerGroupUsers();
    const authorizerGroupUsers = this.getAuthorizerGroupUsers();

    if (this.hasOnlyMaker()) {
      checkerGroupUsers.forEach((u: any) => uniqueUsers.add(u.userId));
      authorizerGroupUsers.forEach((u: any) => uniqueUsers.add(u.userId));
    } else {
      checkerGroupUsers.forEach((u: any) => {
        if (u.userId !== invalidUserId) {
          uniqueUsers.add(u.userId);
        }
      });

      authorizerGroupUsers.forEach((u: any) => {
        if (u.userId !== invalidUserId) {
          uniqueUsers.add(u.userId);
        }
      });
    }

    let totalUsers = 0
    if (this.hasOnlyMaker()) {
      totalUsers = uniqueUsers.size;
    } else if (invalidUserId) {
      totalUsers = uniqueUsers.size; // add back the single maker with extra role
    } else {
      totalUsers = uniqueUsers.size - 1;
    }


    // REQUIRED USERS (1 Maker + others)
    const totalRequired = checker + authorizer;

    if (totalRequired > totalUsers) {
      this.notification.remove();
      this.notification.warning(
        'Invalid Combination',
        `Only ${totalUsers} users are available for the combination.`
      );

      // Only reset in ADD mode
      this.rulesForm.patchValue({
        checkerCount: 0,
        authorizerCount: 0
      }, { emitEvent: false });


      return false;
    }

    // MAKER VALIDATIOn
    const hasMaker =
      this.addUserList?.some((u: any) =>
        u.groupUserRole?.includes('MAKER')
      ) ||
      this.addGUserList?.some((g: any) =>
        g.groupRole?.includes('MAKER')
      );

    if (!hasMaker) {
      this.notification.warning('Error', 'No Maker available');
      return false;
    }

    return true;
  }
  hasOnlyMaker(): boolean {
    const individual = (this.addUserList || []).some((u: any) =>
      u.groupUserRole?.length === 1 &&
      u.groupUserRole.includes('MAKER')
    );

    const group = (this.addGUserList || []).some((g: any) =>
      g.groupRole?.length === 1 &&
      g.groupRole.includes('MAKER')
    );

    return individual || group;
  }
